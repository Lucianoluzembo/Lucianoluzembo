size_t ft_strlen(const char *s)
{
    if (!s) return 0;
    size_t i = 0;
    for (; *s; i++, s++);
    return i;
}

int main(void)
{
	char	*promise = "I will be champion onde day\n";
	char	*repeat = "I promise👌\n";
